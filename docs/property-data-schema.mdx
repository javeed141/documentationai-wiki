---
title: Property Data Schema
---
The property data schema below lists all available fields in a property record. Note that actual property records (actually, most of them) may not have every field present.

<ResponseField name="absenteeOwner" field-type="array" required="false" deprecated="false">
  The owner of a property who does not reside in or otherwise occupy that property.

  `"absenteeOwner": [{"absent": false, "firstDateSeen": "2024-04-11T17:17:53.270Z", "lastDateSeen": "2024-04-11T17:17:53.270Z"}]`

  <Expandable title="absenteeOwner properties" default-open="true">
    <ResponseField name="absent" field-type="boolean" required="false" deprecated="false">
      Is currently an absentee owner.
    </ResponseField>

    <ResponseField name="firstDateSeen" field-type="date" required="false" deprecated="false">
      Earliest date seen of the absentee owner.
    </ResponseField>

    <ResponseField name="lastDateSeen" field-type="date" required="false" deprecated="false">
      Latest date seen of the absentee owner.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="address" field-type="text" required="false" deprecated="false">
  The physical street address for this property.

  `"address": "7206 Walling Ln"`
</ResponseField>

<ResponseField name="appliances" field-type="text" required="false" deprecated="false">
  The appliances included with the property.

  `"appliances": ["Cooktop","Dryer","Dishwasher","Microwave","Refrigerator","Washer"]`
</ResponseField>

<ResponseField name="architecturalStyles" field-type="text" required="false" deprecated="false">
  The architectural style of the property.

  `"architecturalStyles": ["Ranch","Bungalow"]`
</ResponseField>

<ResponseField name="assessedValues" field-type="nested" required="false" deprecated="false">
  The assessed value of the property.

  `"assessedValues": [{"totalAmount": 1745400.0, "improvementsAmount": 1745400.0, "landAmount": 1745400.0, "year": 1985}]`

  <Expandable title="assessedValues properties" default-open="false">
    <ResponseField name="improvementsAmount" field-type="float" required="false" deprecated="false">
      Assessed value for improvements made to the property.
    </ResponseField>

    <ResponseField name="landAmount" field-type="float" required="false" deprecated="false">
      Assessed value for the land/lot of the property.
    </ResponseField>

    <ResponseField name="totalAmount" field-type="float" required="false" deprecated="false">
      Amount value the property is currently assessed for.
    </ResponseField>

    <ResponseField name="year" field-type="integer" required="false" deprecated="false">
      Year value when the property was assessed.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="brandTenant" field-type="string" required="false" deprecated="false">
  The branded tenant that is currently associated with the property.

  `"brandTenant": "Bob Evans Restaurants, LLC"`
</ResponseField>

<ResponseField name="brokers" field-type="nested" required="false" deprecated="false">
  A list of brokers that have represented this property over time.

  `"brokers": {"agent": "Tom Atterson", "company": "Frich & Clare Realty", "dateSeen": "2025-03-28T17:18:49.715Z", ...}`

  <Expandable title="brokers properties" default-open="false">

    <ResponseField name="company" field-type="text" required="false" deprecated="false">
      The company this agent works for (or the company representing this property).
    </ResponseField>

    <ResponseField name="dateSeen" field-type="date" required="false" deprecated="false">
      The date this agent/company was seen representing the property.
    </ResponseField>

    <ResponseField name="emails" field-type="keyword" required="false" deprecated="false">
      A list of email addresses associated with the agent or company.
    </ResponseField>

    <ResponseField name="licenseNumber" field-type="text" required="false" deprecated="false">
      The license number of the agent representing the property.
    </ResponseField>

    <ResponseField name="licenseType" field-type="text" required="false" deprecated="false">
      The license Type of the agent representing the property.
    </ResponseField>

    <ResponseField name="people_key" field-type="text" required="false" deprecated="false">
      The corresponding key that link to Datafiniti's people data record. (contact, demographics, & more)
    </ResponseField>

    <ResponseField name="phones" field-type="keyword" required="false" deprecated="false">
      A phone number associated with the agent or company.
    </ResponseField>

    <ResponseField name="websites" field-type="keyword" required="false" deprecated="false">
      A list of websites associated with the agent or company.

    <ResponseField name="workEmail" field-type="keyword" required="false" deprecated="false">

</ResponseField>
