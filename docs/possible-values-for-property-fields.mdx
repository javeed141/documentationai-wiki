    public static String abbreviateAddress(String address) {
        StringBuilder result = new StringBuilder();
        String[] parts = address.split("(?=[,])|\\s+");
        for (String part : parts) {
            if (part.trim().isEmpty()) continue;
            String sanitized = part.replaceAll("[^a-zA-Z0-9]", "").toUpperCase();
            String abbreviated = abbreviations.getOrDefault(sanitized, part);
            result.append(abbreviated);
            if (part.endsWith(",")) result.append(",");
            result.append(" ");
        }
        return result.toString().trim();
    }

    public static void main(String[] args) {
        String userAddress = "123, Main Street, Alberta, Northwest";
        System.out.println("Original Address: " + userAddress);
        System.out.println("Abbreviated Address: " + abbreviateAddress(userAddress));
    }

def normalize_address(address: str) -> str:
    address = re.sub(r"\s+", " ", address.strip())
    words = address.split()
    unit_indices = [i for i, w in enumerate(words) if w.upper() == "UNIT"]
    street_positions = [i for i, w in enumerate(words) if w.upper() in abbreviations]
    target_index = None
    if street_positions:
        if unit_indices:
            for idx in reversed(street_positions):
                if idx < unit_indices[0]:
                    target_index = idx
                    break
        else:
            target_index = street_positions[-1]
    for i, word in enumerate(words):
        uw = word.upper()
        if uw == "UNIT":
            words[i] = "UNIT"
            continue
        if uw in directions:
            words[i] = directions[uw]
            continue
        if target_index is not None and i == target_index:
            words[i] = abbreviations[uw]
            continue
        words[i] = word.capitalize()
    return " ".join(words)
